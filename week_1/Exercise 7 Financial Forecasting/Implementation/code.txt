public class FinancialForecasting {

    // Method to calculate future value recursively
    public static double calculateFutureValue(double currentValue, double growthRate, int periods) {
        // Base case: If no periods left, return the current value
        if (periods == 0) {
            return currentValue;
        }
        // Recursive case: Calculate future value for one less period and apply growth rate
        return calculateFutureValue(currentValue, growthRate, periods - 1) * (1 + growthRate);
    }

    public static void main(String[] args) {
        double initialValue = 1000.0;  // Initial investment value
        double growthRate = 0.05;      // 5% growth rate
        int periods = 10;              // Number of periods

        double futureValue = calculateFutureValue(initialValue, growthRate, periods);
        System.out.printf("Future Value after %d periods: %.2f%n", periods, futureValue);
    }
}
